import torch
import time
from torch.utils.cpp_extension import load_inline

cpp_src = r"""
#include <torch/extension.h>

torch::Tensor ssm_shared_partial_forward(torch::Tensor u,
                                         torch::Tensor a,
                                         torch::Tensor b,
                                         torch::Tensor c);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ssm_shared_partial_forward", &ssm_shared_partial_forward,
          "Hybrid Shared Memory Partial Mixed Precision SSM");
}
"""

cuda_src = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__global__ void ssm_kernel_shared_partial(
    const half* __restrict__ u,
    const half* __restrict__ a,
    const half* __restrict__ b,
    const half* __restrict__ c,
    half* __restrict__ y,
    int L,
    int N
) {
    extern __shared__ half shmem[];

    half* sh_a = shmem;
    half* sh_b = shmem + blockDim.x;
    half* sh_c = shmem + 2 * blockDim.x;

    int tid = threadIdx.x;
    int n = blockIdx.x * blockDim.x + tid;

    if (n < N) {
        sh_a[tid] = a[n];
        sh_b[tid] = b[n];
        sh_c[tid] = c[n];
    }

    __syncthreads();

    if (n >= N) return;

    // Partial Mixed Precision:
    // input and parameters are FP16
    // state accumulation is FP32
    float h = 0.0f;

    float af = __half2float(sh_a[tid]);
    float bf = __half2float(sh_b[tid]);
    float cf = __half2float(sh_c[tid]);

    for (int t = 0; t < L; t++) {
        float uf = __half2float(u[t * N + n]);

        h = af * h + bf * uf;

        float yf = cf * h;
        y[t * N + n] = __float2half(yf);
    }
}

torch::Tensor ssm_shared_partial_forward(torch::Tensor u,
                                         torch::Tensor a,
                                         torch::Tensor b,
                                         torch::Tensor c) {
    TORCH_CHECK(u.is_cuda(), "u must be CUDA tensor");
    TORCH_CHECK(a.is_cuda(), "a must be CUDA tensor");
    TORCH_CHECK(b.is_cuda(), "b must be CUDA tensor");
    TORCH_CHECK(c.is_cuda(), "c must be CUDA tensor");

    TORCH_CHECK(u.dtype() == torch::kFloat16, "u must be float16");
    TORCH_CHECK(a.dtype() == torch::kFloat16, "a must be float16");
    TORCH_CHECK(b.dtype() == torch::kFloat16, "b must be float16");
    TORCH_CHECK(c.dtype() == torch::kFloat16, "c must be float16");

    u = u.contiguous();
    a = a.contiguous();
    b = b.contiguous();
    c = c.contiguous();

    int L = u.size(0);
    int N = u.size(1);

    auto y = torch::empty_like(u);

    int threads = 256;
    int blocks = (N + threads - 1) / threads;

    size_t shared_mem_size = 3 * threads * sizeof(half);

    ssm_kernel_shared_partial<<<blocks, threads, shared_mem_size>>>(
        reinterpret_cast<half*>(u.data_ptr<at::Half>()),
        reinterpret_cast<half*>(a.data_ptr<at::Half>()),
        reinterpret_cast<half*>(b.data_ptr<at::Half>()),
        reinterpret_cast<half*>(c.data_ptr<at::Half>()),
        reinterpret_cast<half*>(y.data_ptr<at::Half>()),
        L,
        N
    );

    return y;
}
"""

module = load_inline(
    name="ssm_shared_partial_ext",
    cpp_sources=cpp_src,
    cuda_sources=cuda_src,
    extra_cuda_cflags=["-O3"],
    verbose=False
)

def ssm_fp32_reference(u, a, b, c):
    L, N = u.shape
    h = torch.zeros(N, device=u.device, dtype=torch.float32)
    y = torch.empty((L, N), device=u.device, dtype=torch.float32)

    for t in range(L):
        h = a * h + b * u[t]
        y[t] = c * h

    return y

def ssm_fp16_gpu(u, a, b, c):
    L, N = u.shape
    h = torch.zeros(N, device=u.device, dtype=torch.float16)
    y = torch.empty((L, N), device=u.device, dtype=torch.float16)

    for t in range(L):
        h = a * h + b * u[t]
        y[t] = c * h

    return y

def accuracy_metrics(y_ref, y_test):
    y_ref = y_ref.float()
    y_test = y_test.float()

    abs_err = torch.abs(y_ref - y_test)
    rel_err = abs_err / (torch.abs(y_ref) + 1e-8)

    return {
        "mean_abs_error": abs_err.mean().item(),
        "mean_rel_error": rel_err.mean().item(),
        "mse": torch.mean((y_ref - y_test) ** 2).item()
    }

def benchmark_gpu(func, *args, iters=30, warmup=5):
    for _ in range(warmup):
        _ = func(*args)

    torch.cuda.synchronize()

    t0 = time.perf_counter()

    for _ in range(iters):
        _ = func(*args)

    torch.cuda.synchronize()

    t1 = time.perf_counter()
    return (t1 - t0) * 1000 / iters

def benchmark_hybrid_end_to_end(u_cpu, a_cpu, b_cpu, c_cpu, iters=30, warmup=5):
    for _ in range(warmup):
        u_gpu = u_cpu.half().cuda()
        a_gpu = a_cpu.half().cuda()
        b_gpu = b_cpu.half().cuda()
        c_gpu = c_cpu.half().cuda()
        y_gpu = module.ssm_shared_partial_forward(u_gpu, a_gpu, b_gpu, c_gpu)
        y_cpu = y_gpu.cpu()

    torch.cuda.synchronize()

    t0 = time.perf_counter()

    for _ in range(iters):
        u_gpu = u_cpu.half().cuda()
        a_gpu = a_cpu.half().cuda()
        b_gpu = b_cpu.half().cuda()
        c_gpu = c_cpu.half().cuda()

        y_gpu = module.ssm_shared_partial_forward(u_gpu, a_gpu, b_gpu, c_gpu)

        y_cpu = y_gpu.cpu()

    torch.cuda.synchronize()

    t1 = time.perf_counter()
    return (t1 - t0) * 1000 / iters

def main():
    if not torch.cuda.is_available():
        print("CUDA is not available")
        return

    torch.manual_seed(0)

    L = 2048
    N = 1024

    print(f"L={L}, N={N}")

    # CPU side: Hybrid starts here
    u_cpu = torch.randn(L, N, dtype=torch.float32)
    a_cpu = torch.rand(N, dtype=torch.float32) * 0.99
    b_cpu = torch.rand(N, dtype=torch.float32)
    c_cpu = torch.rand(N, dtype=torch.float32)

    # GPU tensors
    u32 = u_cpu.cuda()
    a32 = a_cpu.cuda()
    b32 = b_cpu.cuda()
    c32 = c_cpu.cuda()

    u16 = u32.half()
    a16 = a32.half()
    b16 = b32.half()
    c16 = c32.half()

    print("\n=== Running correctness test ===")

    y_ref = ssm_fp32_reference(u32, a32, b32, c32)
    y_fp16 = ssm_fp16_gpu(u16, a16, b16, c16)
    y_shared_partial = module.ssm_shared_partial_forward(u16, a16, b16, c16)

    torch.cuda.synchronize()

    print("FP16 GPU:", accuracy_metrics(y_ref, y_fp16))
    print("Shared Memory + Partial Mixed:", accuracy_metrics(y_ref, y_shared_partial))

    print("\n=== Latency ===")

    fp16_time = benchmark_gpu(ssm_fp16_gpu, u16, a16, b16, c16)
    shared_partial_time = benchmark_gpu(module.ssm_shared_partial_forward, u16, a16, b16, c16)
    hybrid_time = benchmark_hybrid_end_to_end(u_cpu, a_cpu, b_cpu, c_cpu)

    print(f"FP16 GPU latency:                         {fp16_time:.3f} ms")
    print(f"Shared Memory + Partial Mixed GPU:        {shared_partial_time:.3f} ms")
    print(f"Hybrid CPU-GPU end-to-end latency:        {hybrid_time:.3f} ms")

if __name__ == "__main__":
    main()
